// Yapay Zeka (OCR) Motorunu Daha Hızlı Yapılandır const worker = Tesseract.createWorker({ cacheMethod: 'readOnly', logger: m => { if(m.status === 'recognizing text') { const progress = (m.progress * 100).toFixed(0); document.getElementById('processText').innerText = `Tarama Hızı: %${progress}`; } } }); let followingList = []; let followersList = []; async function initOCR() { await worker.load(); await worker.loadLanguage('eng'); await worker.initialize('eng'); // Sadece alfanümerik karakterlere odaklan (Hızı artırır) await worker.setParameters({ tessedit_char_whitelist: 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789._', }); } initOCR(); function parseNames(text) { if (!text) return []; const regex = /([a-z0-9._]{3,30})/gi; const matches = text.match(regex) || []; const filterOut = ['verified', 'follow', 'following', 'takip', 'et', 'profil', 'resmi', 'post', 'followers', 'posts', 'message', 'suggested']; return [...new Set(matches.map(m => m.toLowerCase()))] .filter(m => m.length > 2 && !filterOut.includes(m) && isNaN(m)); } async function processVideo(type) { const fileInput = document.getElementById(type === 'following' ? 'videoFollowing' : 'videoFollowers'); const file = fileInput.files[0]; const statusDiv = document.getElementById(type === 'following' ? 'statusFollowing' : 'statusFollowers'); const processingStatus = document.getElementById('processingStatus'); if (!file) return; statusDiv.innerText = "Yükleniyor..."; processingStatus.classList.remove('hidden'); const video = document.createElement('video'); video.src = URL.createObjectURL(file); video.muted = true; await new Promise((resolve) => { video.onloadedmetadata = () => resolve(); }); const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d', { alpha: false }); // Performans için alpha kapatıldı const namesFound = new Set(); // TURBO AYARI: Videoyu her 1.5 saniyede bir tara (Hız için ideal) const duration = video.duration; for (let i = 0; i < duration; i += 1.5) { video.currentTime = i; await new Promise(r => video.onseeked = r); // Çözünürlüğü optimize et (Görüntüyü küçültmek işlemi hızlandırır) canvas.width = 400; canvas.height = (video.videoHeight / video.videoWidth) * 400; // Siyah-beyaz filtre (OCR daha hızlı okur) ctx.filter = 'grayscale(100%) contrast(150%)'; ctx.drawImage(video, 0, 0, canvas.width, canvas.height); const { data: { text } } = await worker.recognize(canvas); const names = parseNames(text); names.forEach(name => namesFound.add(name)); statusDiv.innerText = `%${((i/duration)*100).toFixed(0)} tamamlandı...`; } if (type === 'following') followingList = [...namesFound]; else followersList = [...namesFound]; statusDiv.innerText = `Bitti! ${namesFound.size} kişi bulundu.`; processingStatus.classList.add('hidden'); if (followingList.length > 0 && followersList.length > 0) { document.getElementById('analyzeBtn').classList.remove('hidden'); } }
